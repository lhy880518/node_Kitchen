2. 나의 풀이

function bestSet(n, s) {
	var answer = [];
  var remain = s%n;

  //배열에 넣을 일정한 정수 값을 구하고 
  //그 정수값에 배열 맨뒤 부터 나머지를 하나씩 더한다
  if(parseInt(s/n) > 0){
    for(var i= n-1 ; i >= 0 ; i--){
      answer[i] = remain > 0 ? parseInt(s/n)+1 : parseInt(s/n);
      remain--;
    }
  }else{
    answer[0] = -1;
  }

	return answer;
}

// 아래는 테스트로 출력해 보기 위한 코드입니다.
console.log(bestSet(8,7));
