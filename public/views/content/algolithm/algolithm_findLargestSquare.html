function findLargestSquare(board)
{
  var maxlength = 0;
  //1,1부터 탐색을 시작하여 왼쪽, 위쪽, 왼쪽위가 모두 1인 아이들의 중에서 최소값+1로 값을 채운다
  for(var i=0 ; i < board.length ; i++){
    for(var j=0 ; j < board[0].length ; j++){
      board[i][j] = board[i][j]==='O' ? 1 : 0;
      if(i >= 1 && j >= 1 && board[i][j] === 1){
        board[i][j] = Math.min(board[i-1][j-1],board[i][j-1],board[i-1][j]) + 1;
      }
      maxlength = maxlength < board[i][j] ? board[i][j] : maxlength;
    }
  }
	return maxlength * maxlength;
}

//아래 코드는 테스트를 위한 출력 코드 입니다.
//var testBoard = [['X','O','O','O','X'],['X','O','O','O','O'],['X','X','O','O','O'],['X','X','O','O','O'],['X','X','X','X','X']];
