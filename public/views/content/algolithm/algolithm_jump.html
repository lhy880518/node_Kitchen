2. 나의 풀이

function jumpCase(num) {
  //경우의 수가 어떻게 흘러가는지 봅니다.
  //칸이 총 1개 있을때에는 1
  //칸이 총 2개 있을때에는 2
  //칸이 총 3개 있을때에는 3
  //칸이 총 4개 있을때에는 5
  //칸이 총 5개 있을때에는 8
  //....
  //칸이 총 n개 있을때에는 n-1의 경우의 수 + n-2의 경우의 수가 됩니다.
  //재귀함수가 나설 차례입니다.
	var answer = 0;

  //num이 2보다 작다면 num을 리턴 시킵니다.
  if(num <=2){
    return num;
    //그 외에 CASE에는 재귀 호출을 시킵니다.
  }else{
    answer += jumpCase(num-1) + jumpCase(num-2);
  }
	return answer;
}

//아래는 테스트로 출력해 보기 위한 코드입니다.
console.log(jumpCase(5));
