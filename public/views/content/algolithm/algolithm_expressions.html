    2. 나의 풀이

    function expressions(num) {
	var answer = 0;
  //어떻게 할지 생각을 해 보다가
  //입력받은 수 num까지 한단계씩 연속된수로 표현이 가능한지 알아보면 될듯
  //먼저 for 문을 이용해서 1 ~ num까지 한번 훑는다.
  for(var i = 1; i <= num ; i++){
    //1~num까지 1부터 확인한다.
    answer += numCounter(num,i);
  }
	return answer;
}

function numCounter(objnum, num){
  //num을 1씩 증가하면서 하나씩 빼서 0이 되면 최종적으로 카운터를 1 증가시킨다.
  return objnum-num >= 0 ? numCounter(objnum-num, num+1) : (objnum == 0 ? 1 : 0);
}

// 아래는 테스트로 출력해 보기 위한 코드입니다.
console.log(expressions(15));
