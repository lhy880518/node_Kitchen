    function nextBigNumber(n){
  //먼저 10진수를 2진수로 바꾸어 보자
  //console.log((n).toString(2)); 1001110
  //이 바꾼 2진수에서 1을 추출한다.
  //console.log((n).toString(2).match(/1/g)); [ '1', '1', '1', '1' ]
  //이 추출한 1의 갯수를 확인한다.
  //console.log((n).toString(2).match(/1/g).length); 4

	var answer = 0;
  var numflag = n+1;

  while(((n).toString(2)).match(/1/g).length !== ((numflag).toString(2)).match(/1/g).length){
    //n을 1씩 늘려가면서 확인하여 반환한다.
  	numflag ++;
  }
  answer = numflag;
	return answer;
}
