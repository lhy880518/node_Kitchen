2. 나의 풀이

function caesar(s, n) {
  //아스키 코드를 이용하면 쉬울거 같다.
  //대소문자 아스키코드 값은 다음과 같다
  //console.log('A'.charCodeAt());//65
  //console.log('Z'.charCodeAt());//90
  //console.log('a'.charCodeAt());//97
  //console.log('z'.charCodeAt());//122
  //코드번호를 다시 문자로 바꾸는건 아래와 같다.
  //console.log(String.fromCharCode(122));//z

    var result = "";
  //한바퀴에 26이니 26의 나머지만큼만 움직이면 된다. 
  //예를 들면 s가 a이고 n이 27이면 소문자 a ~ z까지 진행 후 한칸 더 가는 것과 동일하기 때문에
  //한바퀴 개념을 없애고 하기 위해서 아래와 같이 더하는 숫자를 만든다.
  //김호영 이란분이 하신거 보고 참고함
  if(n<0){
    //또한 n 값이 -라면 +26을 해서 큰수일때로만 생각해서 처리 
    return caesar(s, n+26)
  }
  var adnum = n%26;

  //내가 한 방법은 다음과 같다.
  //1. 먼저 s의 길이만큼 한바퀴 돌린다.
  for(var i=0; i < s.length ; i++){
    var slistr = s.substring(i,i+1);
    var codenum = slistr.charCodeAt();
    //2. 돌리면서 빈칸이면 그냥 result에 더하고 아니면 아래로 넘어간다.
    if(slistr === " "){
      result += slistr;
    }else{
      //3. 대문자 소문자 구분하여 처리 진행한다.
      if(codenum >= 'A'.charCodeAt() && codenum <= 'Z'.charCodeAt()){
         codenum += adnum;
         result += codenum >= 91 ? String.fromCharCode(codenum - 26) : String.fromCharCode(codenum);
      }else if(codenum >= 'a'.charCodeAt() && codenum <= 'z'.charCodeAt()){
         codenum += adnum;
         result += codenum >= 123 ? String.fromCharCode(codenum - 26) : String.fromCharCode(codenum);
      }
    }
  }
    return result;
}

// 실행을 위한 테스트코드입니다.
console.log('s는 "a B z", n은 4인 경우: ' + caesar("a B z", 4));
